Code Example



<span class="sparkline"
   data-values="[2, 4, 3, 4, 5, 4, 5, 4, 3, 4, 5, 6, 4, 5, 6, 3, 5, 4, 5, 4, 5, 4, 3, 4, 5, 6, 7, 5, 4, 3, 5, 6]"
   data-options='{
   "type": "bar",
   "height": "125",
   "barWidth": 13,
   "colorMap": {
   "7": "#a1a1a1"
   },
   "barSpacing": 2,
   "barColor": "#26B99A"
   }'>
</span>
